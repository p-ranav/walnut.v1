reduce := function(iterable, lambda) { 
  if (iterable.length() == 0) { return 0 }

  i := 1, result := iterable[i - 1];
  while i < iterable.length() {
    result = lambda(result, iterable[i])
    i = i + 1
  }

  return result
};

[1, 2, 3, 4].reduce(function(a, b) { a + b }).print(); // 10
[1, 2, 3, 4].reduce(function(a, b) { a * b }).print(); // 24
