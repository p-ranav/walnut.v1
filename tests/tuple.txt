x = (1, 3.14, "Hello");
x.println();
x[0].println();
x[0] = 2; // does nothing
x.println();

thistuple = ("apple", "banana", "cherry");
for x in thistuple { println(x) }

println(length(thistuple));

thistuple[3] = "orange"; // does nothing
println(thistuple);

z = ("apple",);
z.println();

z = ('a', 'p', 'p', 'l', 'e')
z.println();

// Grouped expression within a tuple
z = ((1 + 2), (3 + 4));
z.println(); // (3, 7)

mutating_tuple = (1, 2, [3, 4, 5]);
mutating_tuple[2][0] = -3;
println(mutating_tuple); // (1, 2, [-3, 4, 5])
