x := 15;

mutate_x := function(n) {
  print("Inside before changing x:", x, n);
  x = 5; // mutates global x
  print("Inside after changing x:", x, n);
};

mutate_x(20); // 15 20
print("Outside:", x); // 5


y := 15;

doesnt_mutate_y := function(n) {
  print("Inside before changing y:", y, n);
  y := 5; // creates a local variable y and sets this to 5; doesn't mutate global y
  print("Inside after changing y:", y, n);
};

doesnt_mutate_y(20); // 15 20
print("Outside:", y); // 15
