(true && false).print() // false
(true || false).print() // true

(true and false).print() // false
(true or false).print() // true

(true and !false).print() // true
(true or !true and true).print()  // (true or (!true and true)) = true
(true or !true and false).print() // (true or (!true and false)) = true

(true and !true and false).print() // (true and !(true and false)) = true - NOT has higher precedence than AND
(true and false and !true).print() // (true and (false and !true)) = false
