(true and false).println() // false
(true or false).println() // true

(true and not false).println() // true
(true or not true and true).println()  // (true or (!true and true)) = true
(true or not true and false).println() // (true or (!true and false)) = true

(true and not true and false).println() // (true and (!true and false)) = false
(true and false and not true).println() // (true and (false and !true)) = false

x = !(true and false)
println(x) // true

x = !(true or false)
println(x) // false
