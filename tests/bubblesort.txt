bubble_sort := function(input) {
  for current in range(length(input) - 1, 0, -1) {
    for i in range(current) {
      if input[i] > input[i+1] {
        temp := input[i];
        input[i] = input[i+1];
        input[i+1] = temp;
      }
    }
  }
  return input;
};

[5, -4, 3, -2, 1].bubble_sort().print();
