nested_whiles := function(i, j, k) {
  i_done := false, j_done := false, k_done := false
  while i < 10 {
    if (i == 9) { i_done = true }
    while j < 10 {
      if (j == 9) { j_done = true }
      while k < 10 {
        if (k == 9) { k_done = true }
        k = k + 1
      }
      j = j + 1
    }
    i = i + 1
  }
  return [i_done, j_done, k_done, "OK"]
};

i := 0, j := 0, k := 0;
print(nested_whiles(i, j, k));
