// Pratt Parsing of Expressions
// Input                         expression->print()
-a * b;                           // ((-a) * b)
!-a;                              // (!(-a))
a + b + c;                        // ((a + b) + c)
a + b - c;                        // ((a + b) - c)
a * b * c;                        // ((a * b) * c)
a * b / c;                        // ((a * b) / c)
a + b / c;                        // (a + (b / c))
a + b * c + d / e - f;            // (((a + (b * c)) + (d / e)) - f)
3 + 4;                            // (3 + 4)
-5 * 5;                           // ((-5) * 5)
5 > 4 == 3 < 4;                   // ((5 > 4) == (3 < 4))
5 < 4 != 3 > 4;                   // ((5 < 4) != (3 > 4))
3 + 4 * 5 == 3 * 1 + 4 * 5;       // ((3 + (4 * 5)) == ((3 * 1) + (4 * 5)))
3 + 4 * 5 == 3 * 1 + 4 * 5;       // ((3 + (4 * 5)) == ((3 * 1) + (4 * 5)))
1 + (2 + 3) + 4;                  // ((1 + (2 + 3)) + 4)
(5 + 5) * 2;                      // ((5 + 5) * 2)
2 / (5 + 5);                      // (2 / (5 + 5))
-(5 + 5);                         // (-(5 + 5))
!(true == true);                  // (!(true == true))
