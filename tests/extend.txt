println(extend(append([1, 2], 3), [4, 5]));

// Dot operation
[1, 2, 3, 4].append(5).extend([6, 7]).println();

[].append("Hello").extend([1, 3.14]).append(true).println();

  .append(5)
  .extend([6, 7])
  .map(function(i) { i * 3 })
  .filter(function(i) { i % 2 == 0 })
  .println();
