x = {"a": 5, "b": 3.14, "c": "Hello", "d": false};
x.println();

z = {
  "value" : 
  { 
    "pi": 3.14, 
    "c": "Hello, 世界",

  } 
};

pi = "pi";
z["value"][pi].println(); // 3.14

z["value"]["func"](2, 3).println(); // 6
