z = {
  "test": 5,
  "value" : 
  { 
    "pi": 3.14, 
    "c": "Hello, 世界", 
    "func": function(a, b) { a * b }
  },
  "test2" : 15
};

z["value"][pi].print(); // 3.14

  print(key, ":", value)
}

// pi : 3.14

  print(pair)

// [c, Hello, 世界]
