// Alias to functions
log := print;
log("Hello");

push := append;
[].push(1).print();
[].push(1).extend([2, 3]).print();

add := function(a, b) { a + b; };
sum := add;
sum(2, 3).print();

proxy := function(message) { print("DEBUG::", message); }
debug := proxy

proxy = function(message) { print("INFO::", message); }
info := proxy

proxy = function(message) { print("ERROR::", message); }
error := proxy

"In function x".debug();
"Thread started".info();
"Segmentation Fault".error();
// DEBUG::In function x
// INFO::Thread started
// ERROR::Segmentation Fault
